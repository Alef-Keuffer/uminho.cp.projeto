calcLine :: NPoint -> (NPoint -> OverTime NPoint)
calcLine [] = const nil
calcLine (p : x) = curry g p (calcLine x)
 where
  g :: (ℚ, NPoint -> OverTime NPoint) -> (NPoint -> OverTime NPoint)
  g (d, f) l = case l of
    [] -> nil
    (x : xs) -> concat . sequenceA [singl . linear1d d x, f xs]
